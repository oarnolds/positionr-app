import * as cheerio from "cheerio";
import type { WebsiteSnapshot } from "./schema";

const PAGES_TO_TRY = [
  "",
  "/diensten",
  "/services",
  "/over-ons",
  "/about",
  "/cases",
  "/referenties",
];

const MAX_CHARS_PER_PAGE = 3000;
const MAX_CHARS_TOTAL = 15_000;

function normalizeBaseUrl(url: string): string {
  let n = url.trim();
  if (!/^https?:\/\//i.test(n)) n = `https://${n}`;
  return n.replace(/\/$/, "");
}

async function scrapeOnePage(url: string): Promise<string | null> {
  try {
    const res = await fetch(url, {
      headers: {
        "User-Agent": "Positionr/1.0 (+https://app.positionr.nl)",
      },
      signal: AbortSignal.timeout(10_000),
    });
    if (!res.ok) return null;
    const html = await res.text();
    const $ = cheerio.load(html);
    $("script, style, noscript").remove();
    const text = $("body").text().replace(/\s+/g, " ").trim();
    if (text.length < 100) return null; // skip vrijwel-lege pagina's
    return text.slice(0, MAX_CHARS_PER_PAGE);
  } catch {
    return null;
  }
}

export async function scrapeForIcp(url: string): Promise<WebsiteSnapshot> {
  const base = normalizeBaseUrl(url);
  const urls = PAGES_TO_TRY.map((p) => base + p);

  const results = await Promise.allSettled(urls.map(scrapeOnePage));
  const pages = results
    .map((r, i) =>
      r.status === "fulfilled" && r.value
        ? { url: urls[i], text: r.value }
        : null
    )
    .filter((x): x is { url: string; text: string } => x !== null);

  if (pages.length === 0) {
    throw new Error(
      `Geen enkele pagina van ${base} kon worden opgehaald. Check URL.`
    );
  }

  // Combineer alle pagina's met header per URL, gecapt op MAX_CHARS_TOTAL
  const combined = pages
    .map((p) => `=== ${p.url} ===\n${p.text}`)
    .join("\n\n")
    .slice(0, MAX_CHARS_TOTAL);

  // Voor backward-compat met WebsiteSnapshot: extraheer korte titel/hero uit homepage
  const homepageHtml = await fetchHomepageMetadata(base);

  return {
    url: base,
    title: homepageHtml.title,
    metaDescription: homepageHtml.metaDescription,
    heroText: homepageHtml.heroText,
    bodyExcerpt: combined,
    scrapedAt: new Date().toISOString(),
  };
}

async function fetchHomepageMetadata(baseUrl: string): Promise<{
  title: string;
  metaDescription: string;
  heroText: string;
}> {
  try {
    const res = await fetch(baseUrl, {
      headers: { "User-Agent": "Positionr/1.0" },
      signal: AbortSignal.timeout(8_000),
    });
    if (!res.ok) return { title: "", metaDescription: "", heroText: "" };
    const html = await res.text();
    const $ = cheerio.load(html);
    return {
      title: $("title").text().trim(),
      metaDescription:
        $('meta[name="description"]').attr("content")?.trim() ?? "",
      heroText: $("h1, h2")
        .slice(0, 3)
        .map((_, el) => $(el).text().trim())
        .get()
        .filter(Boolean)
        .join(" | "),
    };
  } catch {
    return { title: "", metaDescription: "", heroText: "" };
  }
}

"use server";

import { eq } from "drizzle-orm";
import { revalidatePath } from "next/cache";
import { redirect } from "next/navigation";
import { createClient as createSupabaseClient } from "@/lib/supabase/server";
import { db } from "@/lib/db/client";
import { modules, profiles } from "@/lib/db/schema";
import { buildSystemPrompt as buildIcpSystemPrompt } from "@/modules/icp-analyse/prompt";

async function requireAdmin() {
  const supabase = await createSupabaseClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) redirect("/login");
  const [profile] = await db
    .select({ role: profiles.role })
    .from(profiles)
    .where(eq(profiles.id, user.id))
    .limit(1);
  if (profile?.role !== "admin") {
    throw new Error("Geen admin-toegang");
  }
  return user;
}

/** Map slug → code-default-prompt builder (voor "Reset"-knop). */
function getCodeDefaultPrompt(slug: string): string | null {
  switch (slug) {
    case "icp-analyse":
      return buildIcpSystemPrompt();
    default:
      return null; // Andere modules hebben nog geen code-prompt
  }
}

export async function getModulePrompt(slug: string): Promise<{
  slug: string;
  name: string;
  prompt: string;
  hasCodeDefault: boolean;
}> {
  await requireAdmin();
  const [m] = await db
    .select({ slug: modules.slug, name: modules.name, defaultPrompt: modules.defaultPrompt })
    .from(modules)
    .where(eq(modules.slug, slug))
    .limit(1);
  if (!m) throw new Error("Module niet gevonden");
  return {
    slug: m.slug,
    name: m.name,
    prompt: m.defaultPrompt ?? "",
    hasCodeDefault: getCodeDefaultPrompt(slug) !== null,
  };
}

export async function saveModulePrompt(slug: string, prompt: string): Promise<void> {
  await requireAdmin();
  if (prompt.trim().length < 50) {
    throw new Error("Prompt is te kort (minimaal 50 tekens)");
  }
  await db
    .update(modules)
    .set({ defaultPrompt: prompt, updatedAt: new Date() })
    .where(eq(modules.slug, slug));
  revalidatePath(`/admin/prompts/${slug}`);
}

export async function resetModulePrompt(slug: string): Promise<void> {
  await requireAdmin();
  const codeDefault = getCodeDefaultPrompt(slug);
  if (!codeDefault) {
    throw new Error(`Geen code-default beschikbaar voor module '${slug}'`);
  }
  await db
    .update(modules)
    .set({ defaultPrompt: codeDefault, updatedAt: new Date() })
    .where(eq(modules.slug, slug));
  revalidatePath(`/admin/prompts/${slug}`);
}

"use client";

import { useState, useTransition } from "react";
import { Button } from "@/components/ui/button";
import { saveModulePrompt, resetModulePrompt } from "./actions";

export function PromptEditor({
  slug,
  initialPrompt,
  hasCodeDefault,
}: {
  slug: string;
  initialPrompt: string;
  hasCodeDefault: boolean;
}) {
  const [prompt, setPrompt] = useState(initialPrompt);
  const [pending, startTransition] = useTransition();
  const [message, setMessage] = useState<{
    kind: "ok" | "error";
    text: string;
  } | null>(null);

  function handleSave() {
    setMessage(null);
    startTransition(async () => {
      try {
        await saveModulePrompt(slug, prompt);
        setMessage({ kind: "ok", text: "Opgeslagen." });
      } catch (err) {
        setMessage({
          kind: "error",
          text: err instanceof Error ? err.message : "Opslaan mislukt",
        });
      }
    });
  }

  function handleReset() {
    if (!confirm("Prompt resetten naar code-default?")) return;
    setMessage(null);
    startTransition(async () => {
      try {
        await resetModulePrompt(slug);
        // Server action revalidates; eenvoudiger: reload pagina
        window.location.reload();
      } catch (err) {
        setMessage({
          kind: "error",
          text: err instanceof Error ? err.message : "Reset mislukt",
        });
      }
    });
  }

  return (
    <div className="space-y-4">
      <textarea
        value={prompt}
        onChange={(e) => setPrompt(e.target.value)}
        rows={25}
        className="w-full rounded-lg border border-gray-300 p-3 font-mono text-xs"
        spellCheck={false}
      />
      <div className="flex items-center justify-between text-xs text-gray-500">
        <span>{prompt.length.toLocaleString("nl-NL")} tekens</span>
        {message && (
          <span
            className={
              message.kind === "ok" ? "text-green-700" : "text-red-700"
            }
          >
            {message.text}
          </span>
        )}
      </div>
      <div className="flex gap-2">
        <Button onClick={handleSave} disabled={pending}>
          {pending ? "Bezig..." : "Opslaan"}
        </Button>
        {hasCodeDefault && (
          <Button
            variant="outline"
            onClick={handleReset}
            disabled={pending}
          >
            Reset naar code-default
          </Button>
        )}
      </div>
    </div>
  );
}
